Prepare Transaction


                    // Prepare Transaction 
                    $transaction = $nemPhp->prepareTransaction(
                        $amount, // How much XEM to send 
                        $fee, // Put higher fee if you want, otherwise leave it zero so minimum fee will be taken off
                        $receiver, //adress where to send
                        $mosaics = null,   //mosaics
                        $message = null, // message
                        $secure = false // secure message
                    );
